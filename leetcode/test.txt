## 14.longest common prefix

```
class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
		string prefix = "";
		int pl = -1;
        if(strs.size()==0 || strs[0].size() == 0)return "";
		int strlen = strs.size();
		int sl = strs[0].size();
		for(int i =1;i<strlen ;i++){
			for(int j =0;j<sl;j++){
				if(strs[0][j]==strs[i][j]){
					if(j>pl){
						prefix += strs[0][j];
						pl++;
					}
					else{
						continue ;
					}
				}
				else{
					break;
				}
			}
		}
		renturn prefix;
    }
};
```
